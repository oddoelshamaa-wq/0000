async function handleAttendanceClick() { if (!currentUser) return; const btn = document.getElementById('attendanceBtn'); if (btn.classList.contains('btn-disabled')) return; // --- منطق الـ GPS للموبايل فقط --- if (getDeviceType() === "mobile") { const statusText = document.getElementById('attStatusText'); statusText.innerText = "⏳ جاري التحقق من موقعك..."; try { // طلب الحصول على الموقع الحالي const position = await new Promise((resolve, reject) => { navigator.geolocation.getCurrentPosition(resolve, reject, { enableHighAccuracy: true, timeout: 10000, maximumAge: 0 }); }); const userLat = position.coords.latitude; const userLon = position.coords.longitude; // حساب المسافة بين الموظف ومقر الشركة const distance = calculateDistance(userLat, userLon, OFFICE_LAT, OFFICE_LON); if (distance > ALLOWED_RADIUS) { alert(`❌ عذراً! أنت خارج نطاق العمل.\nالمسافة الحالية: ${Math.round(distance)} متر.\nيجب أن تكون داخل الشركة للبصمة من الموبايل.`); statusText.innerText = "خارج النطاق المسموح 🛑"; return; // إيقاف العملية } console.log("GPS Verified! Distance: " + Math.round(distance) + "m"); } catch (error) { let errorMsg = "يرجى تفعيل الـ GPS والسماح للمتصفح بالوصول للموقع."; if (error.code === error.PERMISSION_DENIED) errorMsg = "يجب السماح بالوصول للموقع (Location) للبصمة من الموبايل."; alert(errorMsg); statusText.innerText = "فشل التحقق من الموقع ⚠️"; return; // إيقاف العملية } } // --- نهاية منطق الـ GPS --- // إكمال عملية البصمة العادية (التي تعمل على PC أو بعد نجاح GPS الموبايل) const todayKey = getServerDateKey(); const nowTime = getServerTimeString(); db.ref(`users/${currentUser.username}/attendance/${todayKey}`).once('value', (snap) => { const data = snap.val(); let updateData = { date: todayKey }; if (!data || !data.clockIn) { updateData.clockIn = nowTime; updateData.status = "Present"; updateData.deviceType = getDeviceType(); // تسجيل نوع الجهاز للرقابة alert("✅ تم تسجيل حضورك: " + nowTime); } else if (!data.clockOut) { updateData.clockOut = nowTime; alert("🛑 تم تسجيل انصرافك: " + nowTime); } else { alert("لقد سجلت حضورك وانصرافك بالفعل اليوم! 😎"); return; } db.ref(`users/${currentUser.username}/attendance/${todayKey}`).update(updateData); }); }
